rust
use axum::{Router, routing::get}; use std::net::SocketAddr; async fn hello() -> &'static str { "Hello, World!" } #[tokio::main] async fn main() { let app = Router::new() .route("/", get(hello)); let addr = SocketAddr::from(([127, 0, 0, 1], 3000)); axum::Server::bind(&addr) .serve(app.into_make_service()) .await .unwrap(); }

• ルーティングの追加
